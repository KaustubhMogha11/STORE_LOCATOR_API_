# STORE_LOCATOR_API_
Node/Express/MongoDB API with GeoJSON location field for store locations. Simple vanilla JS frontend using the Mapbox Library
# REQUIREMENTS
  "cors": "^2.8.5",
    "dotenv": "^10.0.0",
    "env": "0.0.2",
    "express": "^4.17.1",
    "mongoose": "^5.12.14",
    "node-geocoder": "^3.27.0"
